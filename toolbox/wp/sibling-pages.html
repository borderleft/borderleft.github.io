List of sibling pages.

<?php
    if ($post->post_parent) {
        $page = $post->post_parent;
    } else {
        $page = $post->ID;
    }

    $children = wp_list_pages(array(
        'child_of' => $page,
        'echo' => '0',
        'title_li' => ''
    ));

    if ($children) {
        echo "<ul>\n".$children."</ul>\n";
    }
?>
