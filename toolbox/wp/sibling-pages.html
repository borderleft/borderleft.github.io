List sibling pages.

<?php
  if ($post->post_parent) {
    $page = $post->post_parent;
  } else {
    $page = $post->ID;
  }

  $children = wp_list_pages([
    'parent' => $page,
    'echo' => false,
    'title_li' => '',
  ]);

  if ($children) {
    echo "<ul>\n".$children."</ul>\n";
  }
?>
