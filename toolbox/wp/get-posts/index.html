Get posts (wordpress)

Get all posts from a category.

<?php
    $postslist = get_posts(array(
        'category' => 1,
        'posts_per_page' => -1
    ));

    if ($postslist) {
        echo '<ul>';
        foreach ($postslist as $post) {
            printf(
                '<li><a href="%s">%s</a></li>',
                esc_url(get_permalink()),
                esc_html(get_the_title())
            );
        }
        echo '</ul>';
        wp_reset_postdata();
    }
?>
