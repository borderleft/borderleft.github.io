Display menu description of menu items.

<?php
/**
 * Display menu description of menu items.
 */
add_filter('walker_nav_menu_start_el', function($item_output, $item, $depth, $args) {
  if (!empty($item->description)) {
    $pattern = '/">'.preg_quote($args->link_before.$item->title, '/').'/';
    $replacement = '">'.$args->link_before
      .'<span class="menu-title">'.esc_html($item->title).'</span>'
      .'<span class="menu-description">'.esc_html($item->description).'</span>';
    $item_output = preg_replace($pattern, $replacement, $item_output, 1);
  }
  return $item_output;
}, 10, 4);
?>

functions.php
