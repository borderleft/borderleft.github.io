Get all posts from a category.

<?php
  $post_list = get_posts([
    'category' => 1,
    'posts_per_page' => -1,
  ]);

  if ($post_list) {
    echo '<ul>';
    foreach ($post_list as $post_item) {
      printf(
        '<li><a href="%s">%s</a></li>',
        esc_url(get_permalink($post_item)),
        esc_html(get_the_title($post_item)),
      );
    }
    echo '</ul>';
  }
?>
